Footer Component

UIKit SwiftUI

The Footer component provides an area at the bottom of the screen for navigation controls, actions, and contextual information.

  • Overview
  • Properties
  • Methods
  • Examples

Overview

The Footer component is a versatile container typically used at the bottom of screens to provide navigation options, primary actions, or supplementary information. Footer components include:

  • Multiple layout styles (standard, toolbar, tabbed, etc.)
  • Support for buttons, text content, and other interactive elements
  • Safe area inset handling for devices with notches or home indicators
  • Customizable background with blur and translucency effects
  • Support for fixed, floating, or auto-hiding behaviors

Footers are commonly used to provide primary actions, navigation controls, or legal/copyright information.

import SuperNovaStorybook

// Create a simple footer with one button
let footer = SNSFooter()
footer.addButton(title: "Continue", style: .primary) {
    // Handle button tap
    navigateToNextScreen()
}

// Add to your screen
screen.addComponent(footer)

Properties

Appearance Properties

Property Type Default Description
style SNSFooterStyle .standard The style of the footer. Options include standard, toolbar, tabbed, minimal, infoBar.
backgroundColor UIColor .systemBackground The background color of the footer.
blurEffect UIBlurEffect.Style? nil The blur effect style applied to the footer background. If set, overrides backgroundColor.
isTranslucent Bool false Whether the footer has a translucent background.
borderStyle SNSFooterBorderStyle .topDivider The border style for the footer. Options include none, topDivider, shadow.
dividerColor UIColor .separator The color of the top divider when borderStyle is .topDivider.
shadowOpacity Float 0.2 The opacity of the shadow when borderStyle is .shadow.

Layout Properties

Property Type Default Description
height CGFloat 56.0 The height of the footer content area (excluding safe area insets).
contentInsets UIEdgeInsets UIEdgeInsets(top: 8, left: 16, bottom: 8, right: 16) The insets for the footer's content relative to its edges.
alignment SNSFooterAlignment .center The alignment of content within the footer. Options include leading, center, trailing, spaceBetween, spaceAround.
respectsSafeArea Bool true Whether the footer respects the safe area insets at the bottom.
buttonSpacing CGFloat 16.0 The horizontal spacing between buttons in the footer.
position SNSFooterPosition .fixed The positioning behavior of the footer. Options include fixed, floating, autoHiding.

Content Properties

Property Type Default Description

attributedText NSAttributedString? nil Optional attributed text displayed in the footer.

textAlignment NSTextAlignment .center The alignment of the text displayed in the footer.
leftAccessoryView UIView? nil Optional custom view displayed on the left side of the footer.
rightAccessoryView UIView? nil Optional custom view displayed on the right side of the footer.

Tab Bar Properties

Property Type Default Description

Examples

Standard Footer with Buttons

import SuperNovaStorybook

func createStandardFooter() -> SNSFooter {
    // Create a standard footer
    let footer = SNSFooter(style: .standard)
    
    // Add a secondary button on the left
    footer.addButton(title: "Cancel", style: .secondary) {
        print("Cancel tapped")
    }
    
    // Add a primary button on the right
    footer.addButton(title: "Continue", style: .primary) {
        print("Continue tapped")
    }
    
    // Set the button alignment to space between (left and right)
    footer.alignment = .spaceBetween
    
    return footer
}

Tabbed Footer

import SuperNovaStorybook

func createTabbedFooter() -> SNSFooter {
    // Create a tabbed footer
    let footer = SNSFooter(style: .tabbed)
    
    // Add tab items
    footer.addTab(
        title: "Home",
        icon: UIImage(systemName: "house")!,
        selectedIcon: UIImage(systemName: "house.fill")!
    ) {
        print("Home tab selected")
    }
    
    footer.addTab(
        title: "Explore",
        icon: UIImage(systemName: "safari")!,
        selectedIcon: UIImage(systemName: "safari.fill")!
    ) {
        print("Explore tab selected")
    }
    
    footer.addTab(
        title: "Notifications",
        icon: UIImage(systemName: "bell")!,
        selectedIcon: UIImage(systemName: "bell.fill")!
    ) {
        print("Notifications tab selected")
    }
    
    footer.addTab(
        title: "Profile",
        icon: UIImage(systemName: "person")!,
        selectedIcon: UIImage(systemName: "person.fill")!
    ) {
        print("Profile tab selected")
    }
    
    // Set a global tab selection handler
    footer.setTabSelectionHandler { index in
        print("Tab selected at index: \(index)")
    }
    
    return footer
}

Footer with Info Text

import SuperNovaStorybook

func createInfoFooter() -> SNSFooter {
    // Create an info bar style footer
    let footer = SNSFooter(style: .infoBar)
    
    // Set the text content
    footer.setText("© 2025 SuperNova Inc. All rights reserved.")
    
    // Customize the text appearance
    footer.setTextAppearance(
        font: UIFont.systemFont(ofSize: 12),
        color: .secondaryLabel,
        alignment: .center
    )
    
    // Set a blur effect background
    footer.setBlurEffect(.systemUltraThinMaterial)
    footer.isTranslucent = true
    
    return footer
}

Floating Action Footer

import SuperNovaStorybook

func createFloatingActionFooter() -> SNSFooter {
    // Create a floating footer
    let footer = SNSFooter(style: .standard)
    footer.position = .floating
    
    // Set appearance for floating style
    footer.contentInsets = UIEdgeInsets(top: 8, left: 16, bottom: 8, right: 16)
    footer.backgroundColor = .systemBackground
    footer.borderStyle = .shadow
    footer.setShadow(opacity: 0.2, radius: 8)
    footer.cornerRadius = 16
    
    // Add a primary action button
    let actionButton = footer.addButton(title: "Add to Cart", style: .primary) {
        print("Add to cart tapped")
    }
    
    // Customize the button
    actionButton.cornerRadius = 8
    actionButton.setIcon(UIImage(systemName: "cart.badge.plus"))
    
    return footer
}

Auto-Hiding Toolbar Footer

import SuperNovaStorybook

func createToolbarFooter() -> SNSFooter {
    // Create a toolbar footer that auto-hides when scrolling
    let footer = SNSFooter(style: .toolbar)
    footer.position = .autoHiding
    
    // Add toolbar items
    let shareButton = SNSButton(icon: UIImage(systemName: "square.and.arrow.up")!, style: .toolbarItem)
    shareButton.onTap = {
        print("Share button tapped")
    }
    footer.addButton(shareButton)
    
    let favoriteButton = SNSButton(icon: UIImage(systemName: "heart")!, style: .toolbarItem)
    favoriteButton.onTap = {
        print("Favorite button tapped")
    }
    footer.addButton(favoriteButton)
    
    let commentButton = SNSButton(icon: UIImage(systemName: "bubble.right")!, style: .toolbarItem)
    commentButton.onTap = {
        print("Comment button tapped")
    }
    footer.addButton(commentButton)
    
    let bookmarkButton = SNSButton(icon: UIImage(systemName: "bookmark")!, style: .toolbarItem)
    bookmarkButton.onTap = {
        print("Bookmark button tapped")
    }
    footer.addButton(bookmarkButton)
    
    // Distribute items evenly
    footer.alignment = .spaceAround
    
    // Set appearance
    footer.setBlurEffect(.systemMaterial)
    footer.isTranslucent = true
    footer.setBorderStyle(.topDivider)
    
    return footer
}

SwiftUI Integration

import SwiftUI
import SuperNovaStorybook

struct FooterDemoView: View {
    @State private var showCheckout = false
    
    var body: some View {
        VStack {
            // Main content
            ScrollView {
                Text("Product Details")
                    .font(.largeTitle)
                    .padding()
                
                // Product content here...
                
                Spacer(minLength: 80) // Space for the footer
            }
            
            // Footer using SNSFooter in SwiftUI
            SNSFooterRepresentable {
                let footer = SNSFooter(style: .standard)
                
                footer.addButton(title: "Add to Wishlist", style: .secondary) {
                    print("Added to wishlist")
                }
                
                let buyButton = footer.addButton(title: "Buy Now", style: .primary) { [weak self] in
                    // Update SwiftUI state
                    self?.showCheckout = true
                }
                buyButton.setIcon(UIImage(systemName: "cart"))
                
                footer.alignment = .spaceBetween
                return footer
            }
            .fixedSize(horizontal: false, vertical: true)
        }
        .sheet(isPresented: $showCheckout) {
            Text("Checkout View")
                .font(.title)
                .padding()
        }
    }
}
